#### Coarse grain 
```cpp

int index = 0;
bool flags[1:N] = ([N]false);
flags[1] = true;


void worker() {

    while(1){
        <
            int mySPot = (index % N) + 1;
            index = index + 1;
        >
        <await(flags[mySpot])>

        // CS

        <flags[mySpot]> = false;
        <flags[mySpot % N +1] = true>

        // Non CS
    }
}
```
